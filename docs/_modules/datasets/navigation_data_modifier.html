class DataModifier:
    """
    A helper class to modify specific values within a given column of a CSV file.

    This class allows loading a CSV, replacing specific values in a chosen column,
    and saving the modified DataFrame to a new file.

    Attributes:
        filename (str): Path to the input CSV file.
        column_to_change (str): Column name in which to perform value replacement.
        initial_value (str): The value in the column to be replaced.
        new_value (str): The new value to replace the initial one.
        data (pd.DataFrame): Loaded data from the CSV file.
    """

    def __init__(self, data_file, column_to_change: str, initial_value, new_value):
        """
        Initializes the DataModifier with file paths and replacement parameters.

        Args:
            data_file (str): Path to the CSV file.
            column_to_change (str): Column name to be modified.
            initial_value (str): Value to search and replace in the column.
            new_value (str): Replacement value.
        """
        self.filename = data_file
        self.column_to_change = column_to_change
        self.data = pd.read_csv(self.filename)
        self.initial_value = initial_value
        self.new_value = new_value

    def change_column_value(self):
        """
        Replaces all occurrences of the initial value with the new value in the specified column.
        """
        self.data[self.column_to_change] = self.data[self.column_to_change].apply(
            lambda x: x.replace(self.initial_value, self.new_value)
        )
